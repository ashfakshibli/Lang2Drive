3. IMAGE SAVING - BGR TO RGB CONVERSION (CRITICAL!)
CARLA outputs images in BGRA format. You MUST convert to RGB before saving:

def save_image_to_disk(image, output_path):
    """Save CARLA image as RGB PNG"""
    try:
        array = np.frombuffer(image.raw_data, dtype=np.uint8)
        array = array.reshape((image.height, image.width, 4))
        # CRITICAL: CARLA uses BGRA - convert BGR to RGB
        array = array[:, :, :3][:, :, ::-1]  # BGR to RGB conversion
        img = Image.fromarray(array)
        img.save(output_path, 'PNG')
        return True
    except Exception as e:
        print(f"[ERROR] Failed to save image: {e}")
        return False

import threading

# Shared state for 5-camera synchronization
images_received = {'front': None, 'front_left': None, 'front_right': None, 'rear': None, 'drone_follow': None}
# WARNING: Do NOT reassign images_received inside main()!
# Use images_received[key] = None to clear, not images_received = {...}
images_lock = threading.Lock()
frame_ready = threading.Event()

def make_camera_callback(camera_name):
    def callback(image):
        with images_lock:
            images_received[camera_name] = image
            if all(img is not None for img in images_received.values()):
                frame_ready.set()
    return callback

# Attach callbacks to all 5 cameras
front_camera.listen(make_camera_callback('front'))
front_left_camera.listen(make_camera_callback('front_left'))
front_right_camera.listen(make_camera_callback('front_right'))
rear_camera.listen(make_camera_callback('rear'))
drone_follow_camera.listen(make_camera_callback('drone_follow'))

# CRITICAL: Camera warm-up - tick world several times BEFORE main loop to initialize cameras

for _ in range(20):  # 20 warm-up ticks
    world.tick()
time.sleep(0.5)  # Give cameras time to start capturing
print("[INFO] Cameras ready")

# In main simulation loop:
world.tick()
if frame_ready.wait(timeout=1.0):
    with images_lock:
        save_image_to_disk(images_received['front'], front_path)
        save_image_to_disk(images_received['front_left'], front_left_path)
        save_image_to_disk(images_received['front_right'], front_right_path)
        save_image_to_disk(images_received['rear'], rear_path)
        save_image_to_disk(images_received['drone_follow'], drone_follow_path)
    frame_number += 1
else:
    # Camera timeout - log warning but continue simulation
    print(f"[WARNING] Frame {frame_number} camera timeout - skipping frame")
frame_ready.clear()
with images_lock:
    images_received['front'] = None
    images_received['front_left'] = None
    images_received['front_right'] = None
    images_received['rear'] = None
    images_received['drone_follow'] = None
